\section{Reproducibility}

\begin{frame}{Reproducibility of Research}
\begin{itemize}
	\item<1-> \textbf<1>{Reproducibility vs Replicability of research?}
	\vspace{2ex}
	\item<2-> \emph{``The confirmation of results and conclusions from one study obtained independently in another'' } \small{(Jasny et al. 2011)}
% reproducibility should in principle be always possible, whereas replicability may not be. 
	\vspace{2ex}		
	\item<3-> \emph{``[T]he independent verification of prior findings''} \small{(Santer et al. 2011)}
	\item<4-> \textbf<4-8>{Levels of Replication}
	\begin{enumerate}
		\item<5-> 1. Re-ask the question
		\item<6-> 2. Re-do the experiment
		\item<7-> 3. Re-analyse the data 
		\item<8-> 4.\textbf<9->{ Reproduce the analysis}
	\end{enumerate}
\end{itemize}
\end{frame}

\item Launched February 2011
\item January 2012 - Project system and Version control integration (git/SVN)

\item October 2014 - direct publishing to shinyapps.io
